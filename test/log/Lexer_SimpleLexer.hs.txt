slexer :: String -> [Token]
slexer = go
  where
    go [] = []
    ------------------------------------------------------------
    -- ē³Źs
    ------------------------------------------------------------
    go ('\n' : rest) =
      TokNewline : go rest
    ------------------------------------------------------------
    -- ŹsāRāüāōāg "-- ..."
    ------------------------------------------------------------
    go ('-' : '-' : rest) =
      let rest' = dropWhile (/= '\n') rest
       in go rest'
    ------------------------------------------------------------
    -- āuāŹābāNāRāüāōāg "{- ... -}"
    ------------------------------------------------------------
    go ('{' : '-' : rest) =
      let rest' = skipBlockComment rest
       in go rest'
    ------------------------------------------------------------
    -- ŗ¾öÆüiāXāyü[āXüj
    ------------------------------------------------------------
    go (' ' : rest) =
      let (spaces, rest') = span (== ' ') rest
       in TokSpace (1 + length spaces) : go rest'
    ------------------------------------------------------------
    -- ĢČÄÜāŖāeāēāŗ
    ------------------------------------------------------------
    go ('\'' : rest) =
      case rest of
        ('\\' : c : '\'' : rest') ->
          TokChar (escapeChar c) : go rest'
        (c : '\'' : rest') ->
          TokChar c : go rest'
        _ ->
          TokUnknown '\'' : go rest
    ------------------------------------------------------------
    -- ĢČÄÜŚ±āŖāeāēāŗ
    ------------------------------------------------------------
    go ('"' : rest) =
      let (str, rest') = readString rest
       in TokString str : go rest'
    ------------------------------------------------------------
    -- ÉöÄÜ
    ------------------------------------------------------------
    go ('-' : c : rest)
      | isDigit c =
          let (digits, rest') = span isDigit rest
           in TokNumber (read (c : digits)) : go rest'
    go (c : rest)
      | isDigit c =
          let (digits, rest') = span isDigit rest
           in TokNumber (read (c : digits)) : go rest'
    ------------------------------------------------------------
    -- Ä»Ģ╩Äq
    ------------------------------------------------------------
    go ('\\' : rest) =
      case rest of
        'c' : 'a' : 's' : 'e' : r
          | not (null r) && not (isIdentChar (head r)) ->
              TokLambdaCase : go r
        '\\':r -> TokOperator "\\" : go r
        _ -> TokSymbol "\\" : go rest
    -- go (c : rest)
    --  | isLetter c =
    --      let (letters, rest') = span isIdentChar rest
    --          name = c : letters

    go (c : rest)
      | isIdentStart c =
          let (letters, rest') = span isIdentChar rest
              name = c : letters
           in classifyIdent name : go rest'
    ------------------------------------------------------------
