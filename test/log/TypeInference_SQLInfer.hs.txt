data SQLInfo = SQLInfo
  { sqlText :: String,
    sqlVars :: [String],
    sqlExprs :: [Expr]
  }

inferSQL :: String -> [Expr] -> SQLInfo
inferSQL sql args =
  SQLInfo
    { sqlText = sql,
      sqlVars = map getVarName args,
      sqlExprs = args
    }

getVarName :: Expr -> String
getVarName (EVar name) = name
getVarName _ = error "SQL arguments must be variables"
